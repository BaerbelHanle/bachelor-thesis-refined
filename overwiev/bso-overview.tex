    \paragraph{Primitive Relationen}

\begin{enumAx}[B]
    \itemT[B1.]{\GSReg(x)}{$x$ ist eine Raumregion}
		
    \itemT[B2.]{\Gsb(x,y)}{$x$ ist eine räumliche Grenze von $y$}
			
    \itemT[B3.]{\Gscoinc(x,y)}{$x$ und $y$ koinzidieren}

    \itemT[B4.]{\Gspart(x,y)}{$x$ ist ein räumlicher Teil von $y$}
		
\end{enumAx}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\paragraph{Definierte Relationen}

\begin{enumAx}[D]      
                        
    \itemT[D1.]{\Gtwodb(x,y) \Ldef \GSReg(y) \wedge \Gsb(x,y)}
            {$x$ ist eine 2-dim.\ Grenze von $y$} 
		 
    \itemT[D2.]{\Gonedb(x,y) \Ldef \GtwoDB(y) \wedge \Gsb(x,y)}
            {$x$ ist eine 1-dim.\ Grenze von $y$}
            
    \itemT[D3.]{\Gzerodb(x,y) \Ldef \GoneDB(y) \wedge \Gsb(x,y)}
            {$x$ ist eine 0-dim.\ Grenze von $y$}             
            
    \itemT[D4.]{\GSB(x) \Ldef \exists y\ \Gsb(x,y)}
            {$x$ ist eine räumliche Grenze}
            
    \itemT[D5.]{\GdDB(x) \Ldef \exists y\ \Gddb(x, y)}
            {$x$ ist eine $\Gd$-dim.\ Grenze}
            
    \itemT[D6.]{\Ggrsb(x,y) \Ldef \Gsb(x,y) \wedge \forall z\ (\Gsb(z,y) \to \Gspart(z,x))}
            {$x$ ist die größte Grenze von $y$}  

    \itemT[D7'.]{-}
            {}

    \itemT[D8'.]{-}
            {}
																							
    \itemTP[D9'.]{$\Geqdim(x,y) \Ldef (\GSReg(x) \wedge \GSReg(y)) \vee \mbox{}$\hfill ($x$ und $y$ haben die selbe Dimension)\\
            \mbox{}\hphantom{$\Geqdim(x,y) \Ldef$}
            $(\GtwoDB(x) \wedge \GtwoDB(y)) \vee
                        (\GoneDB(x) \wedge \GoneDB(y)) \vee
                        (\GzeroDB(x) \wedge \GzeroDB(y))$}{}

    \itemT[D10.]{\Gsppart(x, y) \Ldef \Gspart(x, y) \Land  x \neq y}
            {$x$ ist ein echter Teil von $y$}

     
    \itemT[D11.]{\Gsov(x, y) \Ldef \exists z\  (\Gspart(z, x) \wedge \Gspart(z, y))}
            {$x$ und $y$ überlappen}

     
    \itemTP[D12.]{$\Gsumn(x_{1},\ldots,x_{n},x) \Ldef \forall y\ (\Gsov(y,x) \leftrightarrow \bigvee^{n}_{i=1} \Gsov(y,x_{i}))$\\ \mbox{}}
            {$x$ ist die mereologische Summe von $x_{1}$,\ldots,$x_{n}, n \ge 2$}
            
    \itemTP[D13.]{$\Gintersectn(x_{1},\ldots,x_{n},x)  \Ldef \forall y\ (\Gspart(y,x) \leftrightarrow \bigwedge^{n}_{i=1} \Gspart(y,x_{i}))$\\ \mbox{}}
            {$x$ ist der mereologische Schnitt von $x_{1}$,\ldots,$x_{n}, n \ge 2$}
            
    \itemTP[D14.]{$\Grelcompln(x_{1},\ldots,x_{n},x) \Ldef \mbox{}$\hfill
			              ($x$ ist das rel. Kompl. von $x_{n}$ und $x_{1}$,\ldots,$x_{n-1}, n \ge 2$)\\
			$\hspace*{1em}\bigwedge_{1\leq i < j \leq n} \Geqdim(x_i, x) \wedge
			\forall y\ (\Gspart(y,x) \leftrightarrow \bigwedge^{n-1}_{i=1} \neg\Gsov(y,x_{i}) \wedge \Gspart(y, x_{n}))$}
            {}  

                    
    \itemTP[D15.]{$\Gpartitionn(x_{1},\ldots,x_{n},x) \Ldef \Gsumn(x_{1},\ldots,x_{n},x) \wedge \bigwedge_{1 \leq i < j \leq n} \neg\Gsov(x_i,x_j)$\\ \mbox{}}
    {$x_{1}$,\ldots,$x_{n}$ zerlegen $x$, $n \ge 2$}			

    \itemT[D16.]{\Gtwodhypp(x,y) \Ldef \exists z\ (\Gspart(z,y) \wedge \Gtwodb(x,z))}
            {$x$ ist ein 2-dim.\ Hyperteil von $y$}
    
    \itemTP[D17.]{$\Gonedhypp(x,y) \Ldef \exists z\ ((\Gspart(z,y) \vee \Gtwodhypp(z,y)) \wedge  \Gonedb(x,z)) $\\ \mbox{}}
             {$x$ ist ein 1-dim.\ Hyperteil von $y$} 
    
    \itemTP[D18.]{$\Gzerodhypp(x,y) \Ldef \exists z\ ((\Gspart(z,y) \vee \Gonedhypp(z,y)) \wedge  \Gzerodb(x,z)) $\\ \mbox{}}
             {$x$ ist ein 0-dim.\ Hyperteil von $y$}                  
             
    \itemT[D19.]{\Ghypp(x,y) \Ldef \Gtwodhypp(x,y) \vee \Gonedhypp(x,y) \vee \Gzerodhypp(x,y)}
            {$x$ ist ein Hyperteil von $y$}    

    \itemTP[D20.]{$\Gtangpart(x,y) \Ldef (\Gspart(x,y) \vee \Ghypp(x,y))\ \ \wedge\ \ \mbox{}$
			                         \hfill ($x$ ist ein tangentialer Teil von $y$)\\
			        \hspace*{1em}$\exists x'zz'\ ((\Gspart(x',x) \vee \Ghypp(x',x)) \wedge \mbox{}$\\
							\hspace*{5.2em}$\Gsb(z,y) \wedge (\Gspart(z',z) \vee \Ghypp(z',z)) \wedge \Gscoinc(x',z'))$}
             {}  

    \itemT[D21.]{\Ginpart(x, y) \Ldef (\Gspart(x,y) \vee \Ghypp(x,y)) \Land \neg \Gtangpart(x, y)}
            {\mbox{$x$ ist ein innerer Teil von $y$}}

    \itemT[D22'.]{-}
             {}
             
    \itemT[D23'.]{-}
			      {}	

    \itemTP[D24.]{$\GdDC(x) \Ldef \exists uv\ (\Gpartition(u,v,x)) \wedge  \mbox{}$
                \hfill($x$ ist $\Gd$-dim.\ zusammenhängend)
            \hspace*{1em}$\forall yz\ (\Gpartition(y,z,x) \to 
                    \exists y'z'(\Gddhypp(y',y) \wedge \Gddhypp(z',z) \wedge \Gscoinc(y',z')))$
                    \label{D:dDC}}
                {}

    \itemT[D25.]{\GC(x) \Ldef \GtwoDC(x) \vee \GoneDC(x) \vee \GzeroDC(x)}
                {$x$ ist zusammenhängend} 
                
    \itemT[D26.]{\Gc(x,y) \Ldef \exists z\ (\Gsum(x,y,z) \wedge \GC(z))}
                {$x$ und $y$ sind zusammenhängend} 

    \itemT[D27.]{\Gexc(x,y) \Ldef \Gc(x,y) \wedge \neg\Gsov(x,y)}
                {$x$ und $y$ sind extern zusammenhängend}

    \itemT[D28.]{\GoneCCDd(x) \Ldef \GdDC(x)}
            {$x$ hat eine $\Gd$-dim.\ zusammenhängende Komponente}

    \itemTP[D29.]{$\GnCCDd(x) \Ldef \bigwedge_{i=1}^{n-1} \neg\GiCCDd(x) \wedge
		          \exists x_{1}...x_{n}(\Gpartitionn(x_{1},...,x_{n},x) \wedge \bigwedge_{i=1}^{n} \GoneCCDd(x_{i}))$\\ \mbox{}
							\label{D:nCCDd}}
             {$x$ hat $n$ $\Gd$-dim.\ zusammenhängende Komponenten} 

    \itemT[D30'.]{\GTop(x) \Ldef \GSReg(x) \wedge \GtwoDC(x)}
                {$x$ ist ein Topoid} 
                
    \itemT[D31'.]{\GtwoD(x) \Ldef \GtwoDB(x) \wedge \GoneDC(x)}
                {$x$ ist eine Fläche}  
                            
    \itemT[D32'.]{\GoneD(x) \Ldef \GoneDB(x) \wedge \GzeroDC(x)}
                {$x$ ist eine Linie}
                            
    \itemT[D33'.]{\GzeroD(x) \Ldef \GzeroDB(x) \wedge \neg\exists y\ \Gsppart(y,x)}
        {$x$ ist ein Punkt}              
\end{enumAx}
